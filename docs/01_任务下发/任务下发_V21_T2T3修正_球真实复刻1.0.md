### 2.2 删除 `earthTypeColor()`，改为 `earthNodeFill()`

```js
function earthNodeFill(node) {
  // 节点的渐变 fill：根据亮度等级选择 active 或 locked 渐变
  // lv >= 3 用激活态（中等及以上亮度的节点），lv < 3 用锁定态
  const lv = node.lv || 1;
  return lv >= 3 ? 'url(#rg-active)' : 'url(#rg-locked)';
}

function earthNodeStrokeColor(node) {
  // 节点描边色：5 等级用金色，1-2 等级用更暗的金（与背景拉开但不抢眼）
  const lv = node.lv || 1;
  if (lv >= 5) return 'var(--gold-light)';   // #e6c46a 亮金
  if (lv >= 3) return 'var(--gold)';          // #c8a44a 主金
  return 'var(--gold-dim)';                   // #8a6c2e 暗金
}

function earthNodeStrokeWidth(node) {
  const lv = node.lv || 1;
  return lv >= 5 ? 1.8 : lv >= 3 ? 1.4 : 1.0;
}
```

## 4. T2 修正附加：连线分级（参考 1.0 第 847 行图例）

1.0 有三种连线类型，V2.1 之前全部统一灰色。现在改成：

```js
// 计算连线分级（在 prepareMapData 或 renderEarthSphere 内部）
function linkClass(link) {
  // value 是该连线在所有作品中共现的次数
  if (link.value >= 3) return 'strong';      // 强连接
  if (link.value >= 1) return 'mid';         // 弱连接
  return 'weak';                              // 待点亮（应该不会出现，连线本身就是 value>=1）
}

// 连线渲染
linkLayer.appendChild(svgEl('line', {
  class: `sphere-link sphere-link-${linkClass(link)}`,
  ...
}));
```

CSS：
```css
.sphere-link-strong { stroke: var(--link-strong); stroke-width: 1.5; }   /* rgba(200,164,74,0.7) */
.sphere-link-mid    { stroke: var(--link-mid);    stroke-width: 1.0; }   /* rgba(140,110,60,0.5) */
.sphere-link-weak   { stroke: var(--link-weak);   stroke-width: 0.8; stroke-dasharray: 4,3; } /* rgba(60,54,44,0.6) */
```

视觉验收点（Claude 看截图判断）：

| 项 | 通过标准 |
|---|---------|
| 球的整体感 | 暗背景 + 中心椭圆暖光晕（**像一颗暗夜中的金色行星**）|
| 节点视觉 | 深棕色小圆 + 金色细描边（**像金色框里的黑曜石**）|
| 5 级节点 | 加金色外晕 + glow-filter 模糊（最亮的那种）|
| 1-2 级节点 | 用 rg-locked 更深的渐变，描边用 gold-dim，整体偏暗（**未点亮感**）|
| 节点 vs 球面对比 | 节点深 + 球面光晕亮 → 强对比，节点像浮在金光雾里 |
| 经纬线 | 弱金色辅助线，opacity 0.32/0.24，不喧宾夺主 |
| 连线分级 | 实线粗 / 实线细 / 虚线，三色（强金/中金/暗金）|
| 节点标签 | Noto Serif SC 衬线，颜色 `var(--text-2)`（暖金灰），鎏金质感 |
| **不能出现** | 节点是亮金色 / 球面被金色填充覆盖 / 节点融化进球面 |
