#### JS

```js
function bindReviewAnchors(container) {
  const buttons = container.querySelectorAll('.review-anchor');
  buttons.forEach(btn => {
    btn.addEventListener('click', () => {
      const target = container.querySelector(`#review-${btn.dataset.anchor}`);
      if (target) target.scrollIntoView({ behavior: 'smooth', block: 'start' });
    });
  });

  // IntersectionObserver 自动高亮当前可视区块的锚点
  const sections = container.querySelectorAll('.review-section');
  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        const id = entry.target.id.replace('review-', '');
        buttons.forEach(b => b.classList.toggle('active', b.dataset.anchor === id));
      }
    });
  }, { rootMargin: '-30% 0% -60% 0%' });
  sections.forEach(s => observer.observe(s));
}
```

## 2. T8 宇宙模式（按领域分球 + 紫黑金宇宙背景）

### 2.1 视觉目标

> "紫黑金色的宇宙，不同的小球，每个小球都是一个独立的领域"

视觉设计：
- **背景**：紫黑色径向渐变（中心微紫 → 边缘深黑），模拟宇宙空间
- **星点**：随机散布的金色小亮点（不闪烁，但有不同亮度）
- **星云**：1-2 块大色块（紫色 / 金色低透明度 radial gradient blob）
- **领域球**：每个 domain 一个球，球用 T2 修正后的"深棕渐变 + 金色描边"风格
- **球的位置**：用 Fibonacci sphere 把球分布在画布大尺度位置上
- **球的大小**：根据该领域的作品数缩放（min 28px, max 70px）
- **球之间引力线**：跨领域同名节点 → 两球之间银色细线
- **悬浮态**：hover 球放大 1.15x + 显示该领域名 + 作品数 + 节点数
- **进入态**：click 球 → 切回单球视图，看该领域的所有节点

### 2.2 数据层

```js
// 输入：所有作品，模式（'domain' | 'project'）
// 输出：球数组
function prepareUniverseBalls(articles, mode = 'domain') {
  const groups = {};
  for (const rec of articles) {
    let key, name;
    if (mode === 'domain') {
      key = (rec.analysis?.domain || '未分类').trim() || '未分类';
      name = key;
    } else {
      key = rec.project_id || 'default';
      name = projectNameById[key] || key;
    }
    if (!groups[key]) {
      groups[key] = { ballId: key, name, articles: [], nodeNames: new Set() };
    }
    groups[key].articles.push(rec);
    for (const n of (rec.analysis?.nodes_hit || [])) {
      groups[key].nodeNames.add(n.name);
    }
  }
  return Object.values(groups)
    .map(g => ({
      ballId: g.ballId,
      name: g.name,
      articleCount: g.articles.length,
      nodeCount: g.nodeNames.size,
      nodeNames: g.nodeNames,  // 用于跨球同名节点连线
      articles: g.articles,    // 进入单球视图时用
    }))
    .sort((a, b) => b.articleCount - a.articleCount);  // 大球在前
}

// 跨球同名节点连线
function findCrossBallLinks(balls) {
  const links = [];
  for (let i = 0; i < balls.length; i++) {
    for (let j = i + 1; j < balls.length; j++) {
      const shared = [...balls[i].nodeNames].filter(n => balls[j].nodeNames.has(n));
      if (shared.length > 0) {
        links.push({ source: balls[i].ballId, target: balls[j].ballId, count: shared.length });
      }
    }
  }
  return links;
}
```

function renderStars(layer, count = 30) {
  for (let i = 0; i < count; i++) {
    // 用确定性方式生成位置，避免每次刷新位置抖动
    const t = (i + 0.5) / count;
    const angle = i * Math.PI * (3 - Math.sqrt(5)); // golden angle
    const r = Math.sqrt(t) * 320;
    const cx = 360 + r * Math.cos(angle);
    const cy = 240 + r * Math.sin(angle) * 0.7;  // 略压扁
    const size = 0.8 + Math.random() * 1.4;       // 0.8-2.2
    const opacity = 0.3 + Math.random() * 0.6;    // 0.3-0.9
    layer.appendChild(svgEl('circle', {
      cx, cy, r: size,
      fill: 'var(--gold)',
      opacity,
    }));
  }
}

### 2.5 领域球渲染

```js
function renderUniverseBalls(balls, layer) {
  const cx = 360, cy = 240;
  const baseR = Math.min(cx, cy) - 60;
  
  balls.forEach((ball, i) => {
    // Fibonacci 分布在画布大尺度位置上（不是球面，是 2D 平面分布）
    const t = (i + 0.5) / balls.length;
    const angle = i * Math.PI * (3 - Math.sqrt(5));
    const r = Math.sqrt(t) * baseR;
    const x = cx + r * Math.cos(angle);
    const y = cy + r * Math.sin(angle) * 0.85;  // 略压扁，看起来更"宇宙"
    
    // 球的大小：根据 articleCount 缩放
    const size = 28 + Math.min(42, ball.articleCount * 4);
    
    const g = svgEl('g', { 
      class: 'universe-ball',
      'data-ball-id': ball.ballId,
      transform: `translate(${x},${y})`,
    });
    
    // 球本身：深棕渐变 + 金色描边 + 中心金光晕
    g.appendChild(svgEl('circle', {
      cx: 0, cy: 0, r: size + 8,
      fill: 'url(#rg-glow)',
      opacity: 0.4,
      'pointer-events': 'none',
    }));
    g.appendChild(svgEl('circle', {
      cx: 0, cy: 0, r: size,
      fill: 'url(#rg-active)',
      stroke: 'var(--gold)',
      'stroke-width': 1.6,
      filter: 'url(#ballGlowFilter)',
    }));
    
    // 球名标签
    g.appendChild(svgEl('text', {
      x: 0, y: size + 18,
      'text-anchor': 'middle',
      'font-family': "'Noto Serif SC', serif",
      'font-size': 13,
      'font-weight': 600,
      fill: 'var(--gold-light)',
    }, ball.name));
    
    // 副标签：篇数
    g.appendChild(svgEl('text', {
      x: 0, y: size + 32,
      'text-anchor': 'middle',
      'font-size': 10,
      fill: 'var(--text-3)',
    }, `${ball.articleCount} 篇 · ${ball.nodeCount} 节点`));
    
    // hover 放大 + 描边变亮
    g.addEventListener('pointerenter', () => {
      g.querySelector('circle:nth-child(2)').setAttribute('stroke', 'var(--gold-light)');
      g.querySelector('circle:nth-child(2)').setAttribute('stroke-width', 2.2);
      g.style.transform = `translate(${x}px,${y}px) scale(1.15)`;
      g.style.transformOrigin = `${x}px ${y}px`;
    });
    g.addEventListener('pointerleave', () => {
      g.querySelector('circle:nth-child(2)').setAttribute('stroke', 'var(--gold)');
      g.querySelector('circle:nth-child(2)').setAttribute('stroke-width', 1.6);
      g.style.transform = `translate(${x}px,${y}px) scale(1)`;
    });
    
    // click 进入单球视图
    g.addEventListener('click', () => {
      activeBallId = ball.ballId;
      mapView = 'single';
      saveMapState();
      // 把该领域的作品渲染到单球
      renderEarthSphereForBall(ball);
    });
    
    layer.appendChild(g);
    ball._x = x; ball._y = y; ball._size = size;
  });
}

function renderBallLinks(links, balls, layer) {
  const byId = Object.fromEntries(balls.map(b => [b.ballId, b]));
  links.forEach(link => {
    const a = byId[link.source];
    const b = byId[link.target];
    if (!a || !b) return;
    layer.appendChild(svgEl('line', {
      x1: a._x, y1: a._y, x2: b._x, y2: b._y,
      stroke: 'rgba(200, 200, 230, 0.18)',  // 银色细线
      'stroke-width': Math.min(2, 0.6 + link.count * 0.2),
      'stroke-dasharray': '3 4',
    }));
  });
}
```
